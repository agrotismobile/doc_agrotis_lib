static void showLoading({required BuildContext context, String? title, Color? backgroundColor, Color? colorCircular, Color titleColor = Colors.white}) {
  _overlay ??= OverlayEntry(
    builder: (context) => Material(
      color: Colors.transparent,
      child: ColoredBox(
        color: backgroundColor ?? const Color.fromARGB(80, 29, 29, 29),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          crossAxisAlignment: CrossAxisAlignment.center,
          mainAxisSize: MainAxisSize.min,
          children: [
            Text(
              title ?? '',
              style: TextStyle(
                color: titleColor,
                fontWeight: FontWeight.w400,
                fontSize: AppSize.size(14, context),
              ),
            ),
            const SizedBox(
              height: 8.0,
            ),
            CircularProgressIndicator(valueColor: AlwaysStoppedAnimation(colorCircular ?? AppColors.brand.light)),
          ],
        ),
      ),
    ),
  );
  Overlay.of(context)?.insert(_overlay!);
}
