static Future<DateTime?> openDatePicker(BuildContext context,
    {DateTime? initialDate, DateTime? firstDate, DateTime? endDate}) async {
  return await showDatePicker(
    context: context,
    locale: const Locale('pt', 'BR'),
    initialDate: initialDate ?? DateTime.now(),
    firstDate: firstDate ?? DateTime(2000),
    lastDate: endDate ?? DateTime(3000),
    builder: (context, child) => Theme(
      child: child!,
      data: Theme.of(context).copyWith(
        colorScheme: ColorScheme.light(
          primary: AppColors.brand.dark,
          onPrimary: Colors.white,
          onSurface: AppColors.neutral.darker,
        ),
        textTheme: const TextTheme(
          caption: TextStyle(fontSize: 14.0, fontWeight: FontWeight.w400),
          overline: TextStyle(fontSize: 16.0, fontWeight: FontWeight.w400),
        ),
      ),
    ),
  );
}
