Future<Either<bool, List<T>>> multiGenericModalWithInitValues<T, C>(
  BuildContext context, {
  required List<DefaultMultiGenericModal<T, C>> list,
  required String titleModal,
  List<DefaultMultiGenericModal<T, C>> initialItems = const [],
  int? maxValuesSelected,
}) async {
  bool isSearching = false;
  List<DefaultMultiGenericModal<T, C>> listFiltered = list;
  TextEditingController searchText = TextEditingController();
  bool confirmationButton = false;
  List<int> indexOfItemsSelected = [];

  for (var item in initialItems) {
    for (var itemList in listFiltered) {
      if (item.comparator == itemList.comparator) {
        itemList.isSelected = true;
        if (maxValuesSelected != null) {
          indexOfItemsSelected.add(listFiltered.indexOf(itemList));
        }
      }
    }
  }

  await showDialog<void>(
    barrierColor: Colors.black.withOpacity(0.8),
    context: context,
    builder: (BuildContext context) {
      if (maxValuesSelected != null &&
          (initialItems.length > maxValuesSelected)) {
        return ErrorWidget(
            'maxValuesSelected != null && initialItems.length > maxValuesSelected ');
      }
      return StatefulBuilder(
        builder: (context, setState) {
          void _filter() {
            setState(() {
              listFiltered = list
                  .where((element) => element.title
                      .toLowerCase()
                      .contains(searchText.text.toLowerCase()))
                  .toList();
            });
          }

          void withoutLimitSelected(int index) {
            setState(() {
              listFiltered[index].isSelected = !listFiltered[index].isSelected;
              indexOfItemsSelected.add(index);
            });
          }

          void withLimitSelected(int index) {
            int max = maxValuesSelected ?? 0;
            setState(() {
              if (listFiltered[index].isSelected) {
                indexOfItemsSelected.remove(index);
                listFiltered[index].isSelected = false;
              } else if (!indexOfItemsSelected.contains(index)) {
                indexOfItemsSelected.add(index);
                listFiltered[index].isSelected = true;
              }
              if (indexOfItemsSelected.length > max) {
                listFiltered[indexOfItemsSelected[0]].isSelected = false;
                indexOfItemsSelected.removeAt(0);
              }
            });
          }

          return Container(
            margin: EdgeInsets.only(
              bottom: AppSize.size(7, context),
              top: AppSize.size(9, context),
              left: AppSize.size(8, context),
              right: AppSize.size(8, context),
            ),
            width: AppSize.width(context),
            height: AppSize.height(context),
            decoration: BoxDecoration(
              borderRadius: BorderRadius.circular(5),
              color: Colors.white,
              boxShadow: const [
                BoxShadow(color: Colors.white, spreadRadius: 5),
              ],
            ),
            child: Scaffold(
              backgroundColor: AppColors.neutral.white,
              body: Column(
                children: [
                  Row(
                    children: [
                      Expanded(
                        child: Row(
                          children: [
                            const SizedBox(width: 8),
                            IconButton(
                              icon: Icon(
                                Icons.arrow_back_ios,
                                size: AppSize.size(16, context),
                                color: Colors.black87,
                              ),
                              onPressed: () => Navigator.pop(context),
                            ),
                            Visibility(
                              visible: !isSearching,
                              child: Text(
                                titleModal,
                                style: TextStyle(
                                  fontSize: AppSize.size(16, context),
                                  fontWeight: FontWeight.w500,
                                  color: AppColors.neutral.darker,
                                ),
                              ),
                            ),
                            Visibility(
                              visible: isSearching,
                              child: Expanded(
                                child: TextField(
                                  controller: searchText,
                                  onChanged: (text) => _filter(),
                                  style: TextStyle(
                                    fontSize: AppSize.size(16, context),
                                    fontWeight: FontWeight.w500,
                                    color: AppColors.neutral.darker,
                                  ),
                                  autofocus: true,
                                  decoration: InputDecoration(
                                    hintText: "Pesquisar",
                                    hintStyle: TextStyle(
                                      fontSize: AppSize.size(16, context),
                                      fontWeight: FontWeight.w500,
                                      color: AppColors.neutral.darker,
                                    ),
                                    border: InputBorder.none,
                                  ),
                                ),
                              ),
                            ),
                          ],
                        ),
                      ),
                      Visibility(
                        visible: isSearching,
                        child: IconButton(
                          onPressed: () {
                            setState(() {
                              isSearching = !isSearching;
                              searchText.text = "";
                              _filter();
                            });
                          },
                          icon: const Icon(Icons.close),
                          color: AppColors.neutral.mediumDark,
                        ),
                      ),
                      IconButton(
                        onPressed: () {
                          setState(() {
                            isSearching = !isSearching;
                          });
                        },
                        icon: const Icon(Icons.search),
                        color: AppColors.neutral.mediumDark,
                      ),
                    ],
                  ),
                  Expanded(
                    child: Visibility(
                      visible: listFiltered.isNotEmpty,
                      child: ListView.builder(
                        physics: const BouncingScrollPhysics(),
                        itemBuilder: (context, index) {
                          return MultiModalItem(
                            isSelected: listFiltered[index].isSelected,
                            title: listFiltered[index].title,
                            subtitle: listFiltered[index].subTitle,
                            onTap: () => maxValuesSelected == null
                                ? withoutLimitSelected(index)
                                : withLimitSelected(index),
                          );
                        },
                        itemCount: listFiltered.length,
                      ),
                    ),
                  ),
                  Container(
                    margin:
                        const EdgeInsets.only(bottom: 20, left: 16, right: 16),
                    height: 50,
                    child: ButtonWidget(
                      animationDuration: 300,
                      textButton: 'SELECIONAR',
                      fontSize: (AppSize.width(context) / 24),
                      onPressed: () {
                        confirmationButton = true;
                        Navigator.of(context).pop();
                      },
                    ),
                  ),
                ],
              ),
            ),
          );
        },
      );
    },
  );

  if (confirmationButton) {
    List<DefaultMultiGenericModal<T, C>> itemSelectedOfList =
        listFiltered.where((element) => element.isSelected == true).toList();
    return Right(itemSelectedOfList.map((e) => e.object).toList());
  }
  return const Left(false);
}
