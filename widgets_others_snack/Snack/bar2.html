void bar2(String? message, {bool erro = false, int? seconds, SnackStatus snackStatus = SnackStatus.success, required controller}) {
  var _appController = controller;
  final scaffold = _appController.rootScaffoldMessengerKey.currentState;
  Color _backGroundColor = AppColors.warning.success;
  if (scaffold != null) scaffold.clearSnackBars();
  if (erro) _backGroundColor = AppColors.warning.error;
  if (!erro) {
    switch (snackStatus) {
      case SnackStatus.success:
        _backGroundColor = AppColors.warning.success;
        break;
      case SnackStatus.error:
        _backGroundColor = AppColors.warning.error;
        break;
      case SnackStatus.warning:
        _backGroundColor = AppColors.warning.alert;
        break;
      case SnackStatus.info:
        _backGroundColor = AppColors.warning.info;
        break;
      default:
        _backGroundColor = AppColors.warning.success;
        break;
    }
  }
  final snackBar = SnackBar(
      duration: Duration(seconds: seconds ?? 2),
      content: Container(
        padding: EdgeInsets.symmetric(horizontal: 8),
        child: Padding(
          padding: const EdgeInsets.all(8.0),
          child: Padding(
            padding: const EdgeInsets.symmetric(vertical: 4.0),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              crossAxisAlignment: CrossAxisAlignment.center,
              children: [
                InkWell(
                  onTap: () {},
                  child: Padding(
                    padding: const EdgeInsets.only(bottom: 2.0),
                    child: Icon((erro) ? Icons.warning : (snackStatus == SnackStatus.success ? Icons.check : Icons.warning), color: Colors.white),
                  ),
                ),
                Expanded(
                  child: Container(
                    padding: const EdgeInsets.symmetric(horizontal: 20.0),
                    child: Text2(message.toString(), color: Colors.white, fontWeight: FontWeight.w400, overflow: TextOverflow.visible),
                  ),
                ),
                InkWell(
                  onTap: () {
                    scaffold!.clearSnackBars();
                  },
                  child: Padding(
                    padding: const EdgeInsets.only(bottom: 2.0),
                    child: Icon(Icons.close, color: Colors.white),
                  ),
                ),
              ],
            ),
          ),
        ),
      ),
      backgroundColor: _backGroundColor,
      elevation: 4,
      padding: EdgeInsets.all(8));
  scaffold!.showSnackBar(snackBar);
}
