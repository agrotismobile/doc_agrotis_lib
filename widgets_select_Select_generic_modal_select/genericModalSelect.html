Future<Either<bool, T>> genericModalSelect<T>(BuildContext context,
    {required List<ItemGenericModal<T>> list, required String titleModal, String? initialId}) async {
  ItemGenericModal? selectedItem;
  bool isSearching = false;
  List<ItemGenericModal> listFiltered = list;
  TextEditingController searchText = TextEditingController();

  await showDialog<void>(
    barrierColor: Colors.black.withOpacity(0.8),
    context: context,
    // barrierDismissible: true,
    builder: (BuildContext context) {
      return StatefulBuilder(
        builder: (context, setState) {
          void _filter() {
            setState(() {
              listFiltered = list.where((element) => element.title.toLowerCase().contains(searchText.text.toLowerCase())).toList();
            });
          }

          return Container(
            margin: EdgeInsets.only(
              bottom: AppSize.size(7, context),
              top: AppSize.size(9, context),
              left: AppSize.size(8, context),
              right: AppSize.size(8, context),
            ),
            width: AppSize.width(context),
            height: AppSize.height(context),
            decoration: BoxDecoration(
              borderRadius: BorderRadius.circular(5),
              color: Colors.white,
              boxShadow: const [
                BoxShadow(color: Colors.white, spreadRadius: 5),
              ],
            ),
            child: Scaffold(
              backgroundColor: AppColors.neutral.white,
              body: Column(
                children: [
                  Container(
                    decoration: BoxDecoration(
                      border: Border(
                        bottom: BorderSide(width: 1, color: AppColors.neutral.light),
                      ),
                    ),
                    child: Row(
                      children: [
                        Expanded(
                          child: Row(
                            children: [
                              IconButton(
                                icon: Icon(
                                  Icons.arrow_back_ios,
                                  size: AppSize.size(16, context),
                                  color: Colors.black87,
                                ),
                                onPressed: () => Navigator.of(context).pop(),
                              ),
                              Visibility(
                                visible: !isSearching,
                                child: Text(
                                  titleModal,
                                  style: TextStyle(
                                    fontSize: AppSize.size(16, context),
                                    fontWeight: FontWeight.w500,
                                    color: AppColors.neutral.darker,
                                  ),
                                ),
                              ),
                              Visibility(
                                visible: isSearching,
                                child: Expanded(
                                  child: TextField(
                                    controller: searchText,
                                    onChanged: (text) => _filter(),
                                    style: TextStyle(
                                      fontSize: AppSize.size(16, context),
                                      fontWeight: FontWeight.w500,
                                      color: AppColors.neutral.darker,
                                    ),
                                    autofocus: true,
                                    decoration: InputDecoration(
                                      hintText: "Pesquisar",
                                      hintStyle: TextStyle(
                                        fontSize: AppSize.size(16, context),
                                        fontWeight: FontWeight.w500,
                                        color: AppColors.neutral.darker,
                                      ),
                                      border: InputBorder.none,
                                    ),
                                  ),
                                ),
                              ),
                            ],
                          ),
                        ),
                        Visibility(
                          visible: isSearching,
                          child: IconButton(
                            onPressed: () {
                              setState(() {
                                isSearching = !isSearching;
                                searchText.text = "";
                                _filter();
                              });
                            },
                            icon: const Icon(Icons.close),
                            color: AppColors.neutral.mediumDark,
                          ),
                        ),
                        IconButton(
                          onPressed: () {
                            setState(() {
                              isSearching = !isSearching;
                            });
                          },
                          icon: const Icon(Icons.search),
                          color: AppColors.neutral.mediumDark,
                        ),
                      ],
                    ),
                  ),
                  Container(height: 1, width: double.infinity, color: AppColors.neutral.lightest),
                  Expanded(
                    child: Visibility(
                      visible: list.isNotEmpty,
                      child: ListView.builder(
                        itemBuilder: (context, index) {
                          var value = listFiltered[index];
                          return ItemModal(
                            title: value.title,
                            subTitle: value.subTitle,
                            selected: initialId == null ? false : initialId == value.id,
                            onTap: () {
                              selectedItem = value;
                              Navigator.of(context).pop();
                            },
                          );
                        },
                        itemCount: listFiltered.length,
                      ),
                    ),
                  ),
                ],
              ),
            ),
          );
        },
      );
    },
  );
  if (selectedItem != null) {
    return Right(selectedItem!.object);
  }
  return const Left(false);
}
